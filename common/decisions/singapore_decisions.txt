straits_settlements_act_decision = {
	is_shown = {
		NOT = { has_variable = straits_settlements_act_var }
		exists = c:SGP
		c:SGP = ROOT
		is_subject = yes
		exists = c:BIC
		c:BIC = {
			is_subject = yes
			is_player = no
			NOT = { is_subject_of = ROOT }
			top_overlord = {
				is_player = no
			}
		}
	}
	possible = {
		is_at_war = no
		root.gdp > 2000000
		has_technology_researched = nationalism
		c:BIC = {
			relations:root >= relations_threshold:cordial
			is_at_war = no
			top_overlord = {
				relations:root >= relations_threshold:cordial
				NOT = { has_war_with = ROOT }
			}
		}
	}
	when_taken = {
		set_variable = straits_settlements_act_var
		custom_tooltip = straits_settlements_act_event_tt
		hidden_effect = {
			trigger_event = {
				id = singapore.1
				days = 0
				popup = yes
			}
		}
	}

	ai_chance = {
		base = 1000
	}
}

singapore_crackdown_decision = {
	is_shown = {
		exists = c:SGP
		c:SGP = ROOT
		NOT = { has_variable = singapore_crackdown_in_progress }
		OR = {
			has_law = law_type:law_local_police
			has_law = law_type:law_dedicated_police
			has_law = law_type:law_militarized_police
		}
		
	}
	
	possible = {
		custom_tooltip = {
			text = singapore_crackdown_decision_p
			ROOT = {
				var:r_tension_total >= 0.99
			}
		}
	}
	
	when_taken = {
		custom_tooltip = singapore_crackdown_decision_tt
		set_variable = {
			name = singapore_crackdown_in_progress 
			years = 5
		}
		add_modifier = {
			name = singapore_crackdown
			months = 6
		}
		
		if = {
			limit = {
				AND = {
					has_law = law_type:law_local_police
					institution_investment_level = {
						institution = institution_police
						value <= 2
					}
				}
			}
			random_list = {
				55 = {
					trigger_event = {
						id = singapore.4
						months = 6
						popup = yes
					}			
				}
				45 = {
					trigger_event = {
						id = singapore.5
						months = 6
						popup = yes
					}			
				}		
			}	
		}
		
		else_if = {
			limit = {
				OR = {
					has_law = law_type:law_dedicated_police
					institution_investment_level = {
						institution = institution_police
						value <= 4
					}
				}
			}
			
			random_list = {
				75 = {
					trigger_event = {
						id = singapore.4
						months = 6
						popup = yes
					}			
				}
				25 = {
					trigger_event = {
						id = singapore.5
						months = 6
						popup = yes
					}			
				}		
			}
		}
		
		else_if = {
			limit = {
				OR = {
					has_law = law_type:law_militarized_police
					institution_investment_level = {
						institution = institution_police
						value = 5
					}
				}	
			}
			
			random_list = {
				95 = {
					trigger_event = {
						id = singapore.4
						months = 6
						popup = yes
					}			
				}
				5 = {
					trigger_event = {
						id = singapore.5
						months = 6
						popup = yes
					}			
				}		
			}
		}
	}

	ai_chance = {
		base = 0	
		modifier = {
			trigger = { 
				AND = {
					has_law = law_type:law_local_police
					institution_investment_level = {
						institution = institution_police
						value <= 2
					}
				}
			}
			add = 25
		}		
		modifier = {
			trigger = { 
				OR = {
					has_law = law_type:law_dedicated_police
					institution_investment_level = {
						institution = institution_police
						value <= 4
					}
				}
			}
			add = 55
		}
		modifier = {
			trigger = { 
				OR = {
					has_law = law_type:law_militarized_police
					institution_investment_level = {
						institution = institution_police
						value = 5
					}
				}
			}
			add = 100
		}
	}
}

#Fund Singapore Racial Riot
fund_singapore_riot_north = {
	is_shown = {
		ROOT = {
			any_primary_culture = {
				save_temporary_scope_as = culture_stuff
			}
		}
		exists = c:SGP
		NOT = {c:SGP = ROOT}
		OR = {
			country_has_primary_culture = cu:vietnamese
			country_has_primary_culture = cu:khmer
			country_has_primary_culture = cu:khmu
			country_has_primary_culture = cu:lao
			country_has_primary_culture = cu:thai
			country_has_primary_culture = cu:shan
			country_has_primary_culture = cu:mon
			country_has_primary_culture = cu:burmese
			country_has_primary_culture = cu:karen
			country_has_primary_culture = cu:kachin
		}
		c:SGP = {
			NOT = {has_variable = singapore_riot_cooldown}
			OR = {
				has_modifier = merlion_look_north
				has_modifier = merlion_roar_eternity
			}
			any_scope_pop = {
				this.culture = scope:culture_stuff
			}
		}
	}

	possible = {
		c:SGP = {
			relations:ROOT <= relations_threshold:poor
		}
		gold_reserves >= 10000
	}

	when_taken = {
		set_global_variable = {
			name = singapore_racial_riot_perpetrator
			value = ROOT
		}
		add_treasury = -10000
		c:SGP = {
			custom_tooltip = {
				text = fund_singapore_riot_north.tt
				set_variable = {
					name = singapore_riot_cooldown 
					years = 5
				}
			}
			trigger_event = {
				id = singapore.60
				days = 0
				popup = yes
			}
		}
	}
	
	ai_chance = {
		base = 0	
		modifier = {
			trigger = { gold_reserves >= 10000 }
			add = 1000
		}		
		modifier = {
			trigger = { 
				gold_reserves > 0
				weekly_net_fixed_income > 2000
			}
			add = 100
		}
		modifier = {
			OR = {
				is_diplomatic_play_committed_participant = yes
				is_at_war = yes
			}
			add = -1000
		}
	}
}
